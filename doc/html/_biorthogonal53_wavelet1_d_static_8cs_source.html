43  public static void wavelet53_1d (float[] src, float[] dst, int length)
44  {
45  if (length >= 3) {
46  int half = length >> 1;
47  //if the length is even then subtract 1 from "half"
48  //as there is the same number of low and high frequency values
49  //(Note that "num_lf_values" is equal to "half+1")
50  //For a odd length there is one additional low frequency value (so do not subtract 1)
51  //"half" is one less than "num_lf_values" as we cannot directly
52  //calculate the last value in the for-loop (array bounds)
53  if ((length & 1) == 0)
54  half--;
55  int offsrc = 0;
56  // starting offset for high frequency values (= number of low frequency values)
57  int offdst = half + 1;
58  int num_lf_values = offdst;
59 
60  float last_hf = 0.0f;
61  for (int i = 0; i < half; i++) {
62  //calculate the high frequency value by
63  //subtracting the mean of the immediate neighbors for every second value
64  float hf = src [offsrc + 1] - (src [offsrc] + src [offsrc + 2]) * 0.5f;
65  //smoothing the low frequency value, scale by factor 2
66  //(instead of scaling low frequencies by factor sqrt(2) and
67  //shrinking high frequencies by factor sqrt(2)
68  //and reposition to have all low frequencies on the left side
69  dst [i] = 2.0f * (src [offsrc] + (last_hf + hf) * 0.25f);
70  dst [offdst++] = hf;
71  last_hf = hf;
72  offsrc += 2;
73  }
74  if ((length & 1) == 0) {
75  //the secound last value in the array is our last low frequency value
76  dst [num_lf_values - 1] = src [length - 2];
77  //the last value is a high frequency value
78  //however here we just subtract the previos value (so not really a
79  //biorthogonal 5/3 transformation)
80  //This is done because the 5/3 wavelet cannot be calculated at the boundary
81  dst [length - 1] = src [length - 1] - src [length - 2];
82  } else {
83  dst [num_lf_values - 1] = src [length - 1];
84  }
85  } else {
86  //We cannot perform the biorthogonal 5/3 wavelet transformation
87  //for lengths smaller than 3. We could do a simpler transformation...
88  //Here however, we just copy the values from the source to the destination array
89  for (int i = 0; i < length; i++)
90  dst [i] = src [i];
91  }
92  }

102  public static void wavelet53_1d_inverse (float[] src, float[] dst, int length)
103  {
104  if (length >= 3) {
105  int half = length >> 1;
106  //if the length is even then subtract 1 from "half"
107  //as there is the same number of low and high frequency values
108  //(Note that "num_lf_values" is equal to "half+1")
109  //For a odd length there is one additional low frequency value (so do not subtract 1)
110  //"half" is one less than "num_lf_values" as we cannot directly
111  //calculate the last value in the for-loop (array bounds)
112  if ((length & 1) == 0)
113  half--;
114  // number of low frequency values
115  int num_lf_values = half + 1;
116 
117  float last_lf = 0.5f * src [0] - src [num_lf_values] * 0.25f;
118  float last_hf = src [num_lf_values];
119  //Calculate the first two values outside the for loop (array bounds)
120  dst [0] = last_lf;
121  dst [1] = last_hf + last_lf * 0.5f;
122  for (int i = 1; i < half; i++) {
123  float hf = src [num_lf_values + i];
124  float lf = 0.5f * src [i];
125  //reconstruct the original value by removing the "smoothing"
126  float lf_reconst = lf - (hf + last_hf) * 0.25f;
127  dst [2 * i] = lf_reconst;
128  //add reconstructed low frequency value (left side) and high frequency value
129  dst [2 * i + 1] = lf_reconst * 0.5f + hf;
130  //add other low frequency value (right side)
131  //This must be done one iteration later, as the
132  //reconstructed values is not known earlier
133  dst [2 * i - 1] += lf_reconst * 0.5f;
134  last_hf = hf;
135  last_lf = lf_reconst;
136  }
137 
138  if ((length & 1) == 0) {
139  //restore the last 3 values outside the for loop
140  //adding the missing low frequency value (right side)
141  dst [length - 3] += src [num_lf_values - 1] * 0.5f;
142  //copy the last low frequency value
143  dst [length - 2] = src [num_lf_values - 1];
144  //restore the last value by adding last low frequency value
145  dst [length - 1] = src [length - 1] + src [num_lf_values - 1];
146  } else {
147  //restore the last 2 values outside the for loop
148  //adding the missing low frequency value (right side)
149  dst [length - 2] += src [num_lf_values - 1] * 0.5f;
150  //copy the last low frequency value
151  dst [length - 1] = src [num_lf_values - 1];
152  }
153  } else {
154  //We cannot perform the biorthogonal 5/3 wavelet transformation
155  //for lengths smaller than 3. We could do a simpler transformation...
156  //Here however, we just copy the values from the source to the destination array
157  for (int i = 0; i < length; i++)
158  dst [i] = src [i];
159  }
160  }
161  }
