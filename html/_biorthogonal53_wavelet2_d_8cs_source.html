00062         override protected void TransformRow (float[,] src, float[,] dst, int y, int length)
00063         {
00064             if (length >= MINSIZE) {
00065                 int half = length >> 1;
00066                 //if the length is even then subtract 1 from "half"
00067                 //as there is the same number of low and high-pass values
00068                 //(Note that "numLFValues" is equal to "half+1") 
00069                 //For a odd length there is one additional quency value (so do not subtract 1)
00070                 //"half" is one less than "numLFValues" as we cannot directly
00071                 //calculate the last value in the for-loop (array bounds)
00072                 if ((length & 1) == 0)
00073                     half--;
00074                 int offSrc = 0;
00075                 // starting offset for high-pass values (= number of low-pass values)
00076                 int offdst = half + 1; 
00077                 int numLFValues = offdst;
00078     
00079                 float lastHF = 0.0f;
00080                 for (int i = 0; i < half; i++) {
00081                     //calculate the high-pass value by
00082                     //subtracting the mean of the immediate neighbors for every second value
00083                     float hf = src [offSrc + 1, y] - (src [offSrc, y] + src [offSrc + 2, y]) * MEAN;
00084                     //smoothing the low-pass value, scale by factor 2 
00085                     //(instead of scaling low frequencies by factor sqrt(2) and
00086                     //shrinking high frequencies by factor sqrt(2)
00087                     //and reposition to have all low frequencies on the left side
00088                     dst [i, y] = SCALE * (src [offSrc, y] + (lastHF + hf) * SMOOTH);
00089                     dst [offdst++, y] = hf;
00090                     lastHF = hf;
00091                     offSrc += 2; 
00092                 } 
00093                 if ((length & 1) == 0) {
00094                     //the secound last value in the array is our last low-pass value
00095                     dst [numLFValues - 1, y] = SCALE * src [length - 2, y]; 
00096                     //the last value is a high-pass value
00097                     //however here we just subtract the previos value (so not really a
00098                     //biorthogonal 5/3 transformation)
00099                     //This is done because the 5/3 wavelet cannot be calculated at the boundary
00100                     dst [length - 1, y] = src [length - 1, y] - src [length - 2, y];
00101                 } else {
00102                     dst [numLFValues - 1, y] = SCALE * src [length - 1, y]; 
00103                 }
00104             } else {
00105                 //We cannot perform the biorthogonal 5/3 wavelet transformation
00106                 //for lengths smaller than 3. We could do a simpler transformation...
00107                 //Here however, we just copy the values from the source to the destination array  
00108                 for (int i = 0; i < length; i++)
00109                     dst [i, y] = src [i, y];
00110             }

00182                 //Calculate the first two values outside the for loop (array bounds)

00184                 dst [1, y] = lastHF + lastLF * SCALE_INV;

00187                     float lf = SCALE_INV * src [i, y];
00188                     //reconstruct the original value by removing the "smoothing" 
00189                     float lfReconst = lf - (hf + lastHF) * SMOOTH;
00190                     dst [2 * i, y] = lfReconst;
00191                     //add reconstructed low-pass value (left side) and high-pass value

00195                     //reconstructed values is not known earlier
00196                     dst [2 * i - 1, y] += lfReconst * MEAN;
00197                     lastHF = hf;

00199                 }
